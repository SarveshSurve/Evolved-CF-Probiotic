KO_DCA <- read.csv(file = "CAPKo_24hour_assay.csv")
KO_DCA$Type <- factor(KO_DCA$Type, levels = c('Parent', 'Evolved'))
KO_DCA$Mutation <- factor(KO_DCA$Mutation, labels = c("WT", "italic(cap1)", "italic(cap2)", "italic(cap3)"))

KO_DCA_summary <- KO_DCA %>%
  group_by( Mutation, Type) %>%
  summarise(mean_value = mean(log),
            sd_value = sd(log),
            count = n())
